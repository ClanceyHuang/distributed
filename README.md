architecture
```
- cmd/

    - logservice/
        - distributed.log

    - portal/
        - main.go 
    - registryservice/
        - main.go 
- grades/
    - grades.go 
    - mockdata.go
    - server.go 
- log/
    - client.go
    - server.go 
- portal/
    - handlers.go
    - student.html
    - students.html
    - templates.go
- registry/
    - client.go
    - registration.go
    - server.go
- service/

- runtime/ 
```
